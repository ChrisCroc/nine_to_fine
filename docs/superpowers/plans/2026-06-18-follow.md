# Follow + profil nu — Implementation Plan

> **Mode coach** : Chris code 100 % du code de production, task par task. Claude guide (le pourquoi d'abord) et relit. Les commandes `rails g`, `db:migrate`, `git commit` sont lancées par Chris. Les étapes utilisent des cases (`- [ ]`) pour le suivi.

**Goal:** Suivre / ne plus suivre un utilisateur (self-join, instantané) avec un profil public minimal (username + compteurs + bouton Follow).

**Architecture:** Association self-référentielle `Follow` (deux FK vers `users`), counter_cache `followers_count`/`following_count`, bouton user-specific rendu par le controller en `turbo_stream.replace` (pas de broadcast). Sentinel IDOR sur `destroy`. **Pas de listing de contenu ni d'édition de profil** (dépendent de US18, reportée sem 26+).

**Tech Stack:** Rails 8.1, RSpec + factory_bot, Hotwire (Turbo Streams), Devise.

**Spec de référence:** `docs/superpowers/specs/2026-06-18-follow-design.md`

## Task 1 : Migration `follows` + colonnes counter_cache

**Files:**
- Create: `db/migrate/<ts>_create_follows.rb`
- Create: `db/migrate/<ts>_add_follow_counts_to_users.rb`

- [ ] **Step 1 : Générer le model Follow (squelette + migration)**

Run: `bin/rails g model Follow follower:references followed:references`
⚠️ Le générateur produit `t.references :follower, foreign_key: true` → FK vers une table `followers` **inexistante**. On corrige AVANT de migrer.

- [ ] **Step 2 : Corriger la migration `create_follows`**

```ruby
class CreateFollows < ActiveRecord::Migration[8.1]
  def change
    create_table :follows do |t|
      t.references :follower, null: false, foreign_key: { to_table: :users }
      t.references :followed, null: false, foreign_key: { to_table: :users }
      t.timestamps
    end
    add_index :follows, %i[follower_id followed_id], unique: true
  end
end
```
Pourquoi : colonne `follower_id`/`followed_id`, pas `user_id` → on précise `to_table: :users`. Index unique composite = anti double-follow DB-level.

- [ ] **Step 3 : Générer la migration des compteurs**

Run: `bin/rails g migration AddFollowCountsToUsers followers_count:integer following_count:integer`

- [ ] **Step 4 : Corriger la migration des compteurs (null + default)**

```ruby
class AddFollowCountsToUsers < ActiveRecord::Migration[8.1]
  def change
    add_column :users, :followers_count, :integer, null: false, default: 0
    add_column :users, :following_count, :integer, null: false, default: 0
  end
end
```
Pourquoi : `default: 0` + `null: false` → un user existant a un compteur valide dès la migration.

- [ ] **Step 5 : Migrer**

Run: `bin/rails db:migrate`
Expected: `create_table(:follows)` + `add_column(:users, :followers_count...)` OK, `schema.rb` mis à jour.

```ruby
require "rails_helper"

RSpec.describe Follow, type: :model do
  describe "validations" do
    subject { build(:follow) }

    it { is_expected.to be_valid }

    it "is invalid following the same user twice" do
      follow = create(:follow)
      dup = build(:follow, follower: follow.follower, followed: follow.followed)
      expect(dup).not_to be_valid
    end

    it "is invalid following yourself" do
      user = create(:user)
      expect(build(:follow, follower: user, followed: user)).not_to be_valid
    end

    it "enforces uniqueness at the DB level" do
      follow = create(:follow)
      dup = build(:follow, follower: follow.follower, followed: follow.followed)
      expect { dup.save(validate: false) }.to raise_error(ActiveRecord::RecordNotUnique)
    end
  end

  describe "counter_cache" do
    let(:alice) { create(:user) }
    let(:bob)   { create(:user) }

    it "increments the follower's following_count" do
      expect { create(:follow, follower: alice, followed: bob) }
        .to change { alice.reload.following_count }.by(1)
    end

    it "increments the followed's followers_count" do
      expect { create(:follow, follower: alice, followed: bob) }
        .to change { bob.reload.followers_count }.by(1)
    end
  end

  describe "scoping" do
    let(:alice) { create(:user) }
    let(:bob)   { create(:user) }

    before { create(:follow, follower: alice, followed: bob) }

    it "exposes followed_users on the follower" do
      expect(alice.followed_users).to include(bob)
    end

    it "exposes followers on the followed" do
      expect(bob.followers).to include(alice)
    end

    it "answers following?" do
      expect(alice.following?(bob)).to be true
      expect(bob.following?(alice)).to be false
    end
  end
end
```

- [ ] **Step 4 : Écrire le model `Follow`**

`app/models/follow.rb`
```ruby
class Follow < ApplicationRecord
  belongs_to :follower, class_name: "User", counter_cache: :following_count
  belongs_to :followed, class_name: "User", counter_cache: :followers_count

  validates :follower_id, uniqueness: { scope: :followed_id }
  validate :not_self_follow

  private

  def not_self_follow
    errors.add(:followed_id, "can't follow yourself") if follower_id == followed_id
  end
end
```

- [ ] **Step 5 : Ajouter les associations à `User`**

`app/models/user.rb` (après `has_many :comments`)

```ruby
  has_many :active_follows,  class_name: "Follow", foreign_key: :follower_id, dependent: :destroy
  has_many :followed_users,  through: :active_follows, source: :followed
  has_many :passive_follows, class_name: "Follow", foreign_key: :followed_id, dependent: :destroy
  has_many :followers,       through: :passive_follows, source: :follower

  def following?(user)
    followed_users.exists?(user.id)
  end
```

```ruby
require "rails_helper"

RSpec.describe "Follows", type: :request do
  let(:alice) { create(:user) }
  let(:bob)   { create(:user) }

  describe "POST /users/:user_id/follow" do
    context "when signed in" do
      before { sign_in alice }

      it "creates a follow" do
        expect { post user_follow_path(bob) }.to change(Follow, :count).by(1)
      end

      it "is idempotent" do
        post user_follow_path(bob)
        expect { post user_follow_path(bob) }.not_to change(Follow, :count)
      end
    end

    context "when anonymous" do
      it "redirects to login" do
        post user_follow_path(bob)
        expect(response).to redirect_to(new_user_session_path)
      end
    end
  end

  describe "DELETE /users/:user_id/follow" do
    before { sign_in alice }

    it "destroys own follow" do
      create(:follow, follower: alice, followed: bob)
      expect { delete user_follow_path(bob) }.to change(Follow, :count).by(-1)
    end

    it "does not destroy another user's follow (IDOR sentinel)" do
      carol = create(:user)
      create(:follow, follower: carol, followed: bob)
      expect { delete user_follow_path(bob) }.not_to change(Follow, :count)
    end
  end
end
```

- [ ] **Step 5 : Écrire le controller**

`app/controllers/follows_controller.rb`
```ruby
class FollowsController < ApplicationController
  before_action :set_user

  def create
    current_user.active_follows.find_or_create_by(followed: @user)
    respond_with_button
  end

  def destroy
    current_user.active_follows.find_by(followed: @user)&.destroy
    respond_with_button
  end

  private

  def set_user
    @user = User.find(params[:user_id])
  end

  def respond_with_button
    @user.reload
    respond_to do |format|
      format.turbo_stream {
        render turbo_stream: turbo_stream.replace(
          helpers.dom_id(@user, :follow),
          partial: "users/follow_button",
          locals: { user: @user }
        )
      }
      format.html { redirect_to user_path(@user) }
    end
  end
end
```
⚠️ `authenticate_user!` est **déjà global** (`ApplicationController` ligne 9) → pas besoin de le redéclarer. Le request spec « anonyme → redirect login » le prouve.
⚠️ `@user.reload` : sinon le partial ré-affiche l'ancien `followers_count` (staleness mémoire vs counter_cache SQL).
⚠️ Le `format.turbo_stream` rend le partial `users/follow_button` → il doit exister (Task 4). Le request spec ci-dessus ne teste que les compteurs DB, pas le rendu → il passe avant Task 4. Pour le test manuel navigateur, faire Task 4 d'abord.
